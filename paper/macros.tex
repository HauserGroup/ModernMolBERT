% ============================================================
%  TABLE COLUMN TYPES (used by the embedder comparison table)
% ============================================================
% L{width} — ragged-right paragraph column
% K{width} — centred paragraph column (K avoids clash with standard c)
\newcolumntype{L}[1]{>{\RaggedRight\arraybackslash}p{#1}}
\newcolumntype{K}[1]{>{\centering\arraybackslash}p{#1}}
